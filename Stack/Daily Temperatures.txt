class Solution {
    public  static int[] dailyTemperatures(int[] temperatures) {
        int n = temperatures.length;
        int[] result = new int[n];
        Stack<Integer> stack = new Stack<>();
        
        for (int i = 0; i < n; i++) {
            // While the stack is not empty and the current temperature is warmer
            // than the temperature at the index stored at the top of the stack
            while (!stack.isEmpty() && temperatures[i] > temperatures[stack.peek()]) {
                int index = stack.pop();
                result[index] = i - index; // Calculate the number of days
            }
            stack.push(i); // Push the current index onto the stack
        }
        
        // The indices left in the stack do not have a warmer day in the future
        while (!stack.isEmpty()) {
            result[stack.pop()] = 0;
        }
        
        return result;
    }
}
