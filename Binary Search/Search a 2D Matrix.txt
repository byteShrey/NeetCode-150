class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        for(int [] row : matrix ){
            if(binarySearch(row,target)) return true;
        }
        return false;
    }

    private boolean binarySearch(int []arr,int target){
        int low =0;
        int high = arr.length-1;

        

        while(low<=high){
            int mid = low + ((high-low)>>1);
            if(arr[mid]==target){
                return true;
            }else if(arr[mid]<target){
                low= mid+1;
            }else {
                high = mid-1;
            }
        }
        return false;
    }
}
